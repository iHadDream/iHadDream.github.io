什么是Promise

发表于 2017-04-17 | 分类于 study |

原生AJAX

var url = 'http://baidu.com';
var result;

new XHR = new XMLHttpRequest();

XHR.open(GET,url,true);

XHR.send();

XHR.onreadystatechange=function(){
    if(XHR.readyState==4 && XHR.status == 200){
        result =  XHR.response;
        console.log(result);
    }
}

如果要利用这个请求的返回结果，就得在上面的onreadystatechange函数中嵌套一个新的ajax请求

当出现很多请求时，出现很多很多回调函数，成为 回调地狱

Promise 解决了这个问题

当然它还能够将数据请求与数据处理分开

promise对象有三种状态

1.pending: 等待中，暂时还没有结果

2.resolved: 已经完成，收到想要的结果，可以继续往下执行

3.rejected: 也得到结果，但是结果并非我们所愿，所以拒绝执行

状态只能从pending改变到resolved/rejected

promise构造函数中第一个参数时一个函数，来执行promise状态变化

new Promise(function(resolved,reject){
    if(true){resolved()};
    if(false){reject()};
})

Promise对象中的then方法，接受构造函数中处理的状态变化，分别对应执行，有2个参数，第一个参数接受resolved状态的执行，第二个参数接受rejected状态的执行

function fn(num){
    return new Promise(function(resolved,reject){
                if(typeof num == 'number'){
                    resolve();
                }else{
                    reject();
                }
            }).then(function(){
                console.log("这是一个数字");
            },function(){
                console.log("这不是一个数字");
            })
}

fn("aaasdsa");
fn(1244);

then方法也会返回一个promise对象，因此可以链式调用then方法，解决回调地狱的主要方式

function fn(num){
    return new Promise(function(resolved,reject){
                if(typeof num == 'number'){
                    resolve();
                }else{
                    reject();
                }
            }).then(function(){
                console.log("这是一个数字");
            }).then(null,function(){
                console.log("这不是一个数字")
            })
}

fn("aaasdsa");
fn(1244);

Promise中的数据传递

var fn = function(num){
      return  new Promise(function(resolve,reject){
        if(typeof num == "number"){
            resolve(num);
        }else{
            reject('TypeofError');
        }
    })
}
fn(2).then(function(num){
    console.log("first:"+num);
    return num+1;
}).then(function(num){
    console.log("second:"+num);
    return num+1;
}).then(function(num){
    console.log("third:"+num);
    return num+1;
});

利用Promise的知识，对最开始的ajax进行一个简单的封装
var url=’https://hq.tigerbrokers.com/fundamental/finance_calendar/getType/2017-02-26/2017-06-10‘;
//封装一个get请求
function getJSON(url){
return new Promise(function(resolve,reject){
var XHR = new XMLHttpRequest();
XHR.open(‘GET’,url,true);
XHR.send();

        XHR.onreadychange = function(){
            if(XHR.readystate==4){
                if(XHR.status==200){
                    try{
                        var response =  JSON.parse(XHR.responseText);
                        resolve(response);
                    }catch(e){
                        reject(e);
                    }
                }else{
                    reject(new Error(XHR.statusText));
                }
            }
        }
    })
}
getJSON(url).then(resp=>console.log(resp));

为了健壮性，处理了很多可能出现的异常，总之，就是正确的返回结果，就resolve一下，错误的返回结果，就reject一下。并且利用上面的参数传递的方式，将正确结果或者错误信息通过他们的参数传递出来。
